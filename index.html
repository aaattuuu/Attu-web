        function guardarDatos(event, producto) {

            event.preventDefault();

            const form = event.target;

            const nombre = form.nombre.value.trim();

            const cantidad = form.cantidad.value;

            const talla = form.talla.value;


            if (!nombre || !cantidad || !talla) {

                alert('Por favor, completa todos los campos.');

                return;

            }


            const datos = `${nombre},${producto},${cantidad},${talla}\n`;


            try {

                let pedidos = localStorage.getItem('pedidos') || '';

                pedidos += datos;

                localStorage.setItem('pedidos', pedidos);

                alert('Pedido registrado con éxito');

                form.reset();

            } catch (error) {

                alert('Hubo un error al guardar los datos.');

                console.error(error);

            }
